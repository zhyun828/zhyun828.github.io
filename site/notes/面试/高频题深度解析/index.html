  uint16_t length;      // ж•°жҚ®й•ҝеәҰ
  uint8_t data[256];
  uint16_t crc;

enum RxState {
  STATE_IDLE,
  STATE_HEADER,
  STATE_CMD,
  STATE_LEN,
  STATE_DATA,
  STATE_CRC_H,
  STATE_CRC_L,
  STATE_COMPLETE
};

int rx_process(uint8_t byte) {
  static RxState state = STATE_IDLE;
  static Frame frame;
  static int index = 0;

  switch(state) {
    case STATE_IDLE:
      if (byte == 0xAA) {
        state = STATE_HEADER;
      }
      break;

    case STATE_HEADER:
      frame.header = byte;
      state = STATE_CMD;
      break;

    case STATE_CMD:
      frame.cmd = byte;
      state = STATE_LEN;
      index = 0;
      break;

    case STATE_LEN:
      frame.length = (frame.length << 8) | byte;
      index++;
      if (index >= 2) {
        state = STATE_DATA;
        index = 0;
      }
      break;

    case STATE_DATA:
      frame.data[index++] = byte;
      if (index >= frame.length) {
        state = STATE_CRC_H;
      }
      break;

    case STATE_CRC_H:
      frame.crc = byte << 8;
      state = STATE_CRC_L;
      break;

    case STATE_CRC_L:
      frame.crc |= byte;
      // йӘҢиҜҒCRC
      if (verify_crc(&frame)) {
        state = STATE_COMPLETE;
        return 1;  // её§е®Ңж•ҙ
      } else {
        state = STATE_IDLE;  // й”ҷиҜҜпјҢйҮҚж–°жүҫеӨҙ
        return -1;
      }
  }

  return 0;  // з»§з»ӯжҺҘж”¶
